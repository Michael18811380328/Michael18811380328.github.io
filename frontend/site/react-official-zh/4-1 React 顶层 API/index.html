    React.PureComponent¶

    React.PureComponent дёҺ React.Component еҫҲзӣёдјјгҖӮдёӨиҖ…зҡ„еҢәеҲ«еңЁдәҺ React.Component е№¶жңӘе®һзҺ° shouldComponentUpdate()пјҢиҖҢ React.PureComponent дёӯд»Ҙжө…еұӮеҜ№жҜ” prop е’Ң state зҡ„ж–№ејҸжқҘе®һзҺ°дәҶиҜҘеҮҪж•°гҖӮ

    жіЁж„Ҹ

    React.PureComponent дёӯзҡ„ shouldComponentUpdate() д»…дҪңеҜ№иұЎзҡ„жө…еұӮжҜ”иҫғгҖӮеҰӮжһңеҜ№иұЎдёӯеҢ…еҗ«еӨҚжқӮзҡ„ж•°жҚ®з»“жһ„пјҢеҲҷжңүеҸҜиғҪеӣ дёәж— жі•жЈҖжҹҘж·ұеұӮзҡ„е·®еҲ«пјҢдә§з”ҹй”ҷиҜҜзҡ„жҜ”еҜ№з»“жһңгҖӮд»…еңЁдҪ зҡ„ props е’Ң state иҫғдёәз®ҖеҚ•ж—¶пјҢжүҚдҪҝз”Ё React.PureComponentпјҢжҲ–иҖ…еңЁж·ұеұӮж•°жҚ®з»“жһ„еҸ‘з”ҹеҸҳеҢ–ж—¶и°ғз”Ё forceUpdate() жқҘзЎ®дҝқз»„д»¶иў«жӯЈзЎ®ең°жӣҙж–°гҖӮдҪ д№ҹеҸҜд»ҘиҖғиҷ‘дҪҝз”Ё immutable еҜ№иұЎеҠ йҖҹеөҢеҘ—ж•°жҚ®зҡ„жҜ”иҫғгҖӮ

    React.forwardRef¶

    React.forwardRef дјҡеҲӣе»әдёҖдёӘReactз»„д»¶пјҢиҝҷдёӘз»„д»¶иғҪеӨҹе°Ҷе…¶жҺҘеҸ—зҡ„ ref еұһжҖ§иҪ¬еҸ‘еҲ°е…¶з»„д»¶ж ‘дёӢзҡ„еҸҰдёҖдёӘз»„д»¶дёӯгҖӮиҝҷз§ҚжҠҖжңҜе№¶дёҚеёёи§ҒпјҢдҪҶеңЁд»ҘдёӢдёӨз§ҚеңәжҷҜдёӯзү№еҲ«жңүз”Ёпјҡ

    • иҪ¬еҸ‘ refs еҲ° DOM з»„д»¶
    • еңЁй«ҳйҳ¶з»„д»¶дёӯиҪ¬еҸ‘ refs

    React.forwardRef жҺҘеҸ—жёІжҹ“еҮҪж•°дҪңдёәеҸӮж•°гҖӮReact е°ҶдҪҝз”Ё props е’Ң ref дҪңдёәеҸӮж•°жқҘи°ғз”ЁжӯӨеҮҪж•°гҖӮжӯӨеҮҪж•°еә”иҝ”еӣһ React иҠӮзӮ№гҖӮ

    const FancyButton = React.forwardRef((props, ref) => (  
  <button ref={ref} className="FancyButton">    
    {props.children}
  </button>
));

// You can now get a ref directly to the DOM button:
const ref = React.createRef();
<FancyButton ref={ref}>Click me!</FancyButton>;

    еңЁдёҠиҝ°зҡ„зӨәдҫӢдёӯпјҢReact дјҡе°Ҷ <FancyButton ref={ref}> е…ғзҙ зҡ„ ref дҪңдёәз¬¬дәҢдёӘеҸӮж•°дј йҖ’з»ҷ React.forwardRef еҮҪж•°дёӯзҡ„жёІжҹ“еҮҪж•°гҖӮиҜҘжёІжҹ“еҮҪж•°дјҡе°Ҷ ref дј йҖ’з»ҷ <button ref={ref}> е…ғзҙ гҖӮ

    еӣ жӯӨпјҢеҪ“ React йҷ„еҠ дәҶ ref еұһжҖ§д№ӢеҗҺпјҢref.current е°ҶзӣҙжҺҘжҢҮеҗ‘ <button> DOM е…ғзҙ е®һдҫӢгҖӮ

    ж¬ІдәҶи§ЈжӣҙеӨҡзӣёе…ідҝЎжҒҜпјҢиҜ·еҸӮйҳ… refs иҪ¬еҸ‘гҖӮ

    React.lazy¶

    React.lazy() е…Ғи®ёдҪ е®ҡд№үдёҖдёӘеҠЁжҖҒеҠ иҪҪзҡ„з»„д»¶гҖӮиҝҷжңүеҠ©дәҺзј©еҮҸ bundle зҡ„дҪ“з§ҜпјҢе№¶е»¶иҝҹеҠ иҪҪеңЁеҲқж¬ЎжёІжҹ“ж—¶жңӘз”ЁеҲ°зҡ„з»„д»¶гҖӮ

    дҪ еҸҜд»ҘеңЁд»Јз ҒеҲҶеүІж–ҮжЎЈдёӯеӯҰд№ еҰӮдҪ•дҪҝз”Ёе®ғгҖӮжҹҘйҳ…жӯӨж–Үз« еҸҜд»ҘдәҶи§ЈжӣҙеӨҡз”Ёжі•з»ҶиҠӮгҖӮ

    // иҝҷдёӘз»„д»¶жҳҜеҠЁжҖҒеҠ иҪҪзҡ„
const SomeComponent = React.lazy(() => import('./SomeComponent'));

    иҜ·жіЁж„ҸпјҢжёІжҹ“ lazy з»„д»¶дҫқиө–иҜҘз»„д»¶жёІжҹ“ж ‘дёҠеұӮзҡ„ <React.Suspense> з»„д»¶гҖӮиҝҷжҳҜжҢҮе®ҡеҠ иҪҪжҢҮзӨәеҷЁпјҲloading indicatorпјүзҡ„ж–№ејҸгҖӮ

    жіЁж„Ҹ

    дҪҝз”Ё React.lazy зҡ„еҠЁжҖҒеј•е…Ҙзү№жҖ§йңҖиҰҒ JS зҺҜеўғж”ҜжҢҒ PromiseгҖӮеңЁ IE11 еҸҠд»ҘдёӢзүҲжң¬зҡ„жөҸи§ҲеҷЁдёӯйңҖиҰҒйҖҡиҝҮеј•е…Ҙ polyfill жқҘдҪҝз”ЁиҜҘзү№жҖ§гҖӮ

    React.Suspense¶

    React.Suspense еҸҜд»ҘжҢҮе®ҡеҠ иҪҪжҢҮзӨәеҷЁпјҲloading indicatorпјүпјҢд»ҘйҳІе…¶з»„д»¶ж ‘дёӯзҡ„жҹҗдәӣеӯҗз»„д»¶е°ҡжңӘе…·еӨҮжёІжҹ“жқЎд»¶гҖӮзӣ®еүҚпјҢжҮ’еҠ иҪҪз»„д»¶жҳҜ <React.Suspense> ж”ҜжҢҒзҡ„**е”ҜдёҖ**з”ЁдҫӢпјҡ

    // иҜҘз»„д»¶жҳҜеҠЁжҖҒеҠ иҪҪзҡ„
const OtherComponent = React.lazy(() => import('./OtherComponent'));

function MyComponent() {
  return (
    // жҳҫзӨә <Spinner> з»„д»¶зӣҙиҮі OtherComponent еҠ иҪҪе®ҢжҲҗ
    <React.Suspense fallback={<Spinner />}>
      <div>
        <OtherComponent />
      </div>
    </React.Suspense>
  );
}

    е®ғе·Іиў«ж”¶еҪ•еңЁдәҶжҲ‘д»¬зҡ„д»Јз ҒеҲҶеүІжҢҮеҚ—дёӯгҖӮиҜ·жіЁж„ҸпјҢlazy з»„д»¶еҸҜд»ҘдҪҚдәҺ Suspense з»„д»¶ж ‘зҡ„ж·ұеӨ„вҖ”вҖ”е®ғдёҚеҝ…еҢ…иЈ…ж ‘дёӯзҡ„жҜҸдёҖдёӘе»¶иҝҹеҠ иҪҪз»„д»¶гҖӮжңҖдҪіе®һи·өжҳҜе°Ҷ <Suspense> зҪ®дәҺдҪ жғіеұ•зӨәеҠ иҪҪжҢҮзӨәеҷЁпјҲloading indicatorпјүзҡ„дҪҚзҪ®пјҢиҖҢ lazy() еҲҷеҸҜиў«ж”ҫзҪ®дәҺд»»дҪ•дҪ жғіиҰҒеҒҡд»Јз ҒеҲҶеүІзҡ„ең°ж–№гҖӮ

    иҷҪз„¶зӣ®еүҚе°ҡжңӘж”ҜжҢҒе…¶е®ғзү№жҖ§пјҢдҪҶжңӘжқҘжҲ‘д»¬и®ЎеҲ’и®© Suspense ж”ҜжҢҒеҢ…жӢ¬ж•°жҚ®иҺ·еҸ–еңЁеҶ…зҡ„жӣҙеӨҡеңәжҷҜгҖӮдҪ еҸҜд»ҘеңЁ roadmap дёӯдәҶи§Јзӣёе…ідҝЎжҒҜгҖӮ

    жіЁж„Ҹ:

    React.lazy() е’Ң <React.Suspense> е°ҡжңӘеңЁ ReactDOMServer дёӯж”ҜжҢҒгҖӮиҝҷжҳҜе·ІзҹҘй—®йўҳпјҢе°ҶдјҡеңЁжңӘжқҘи§ЈеҶігҖӮ
