дҪҝз”Ё Effect Hook¶

Effect Hook еҸҜд»Ҙи®©дҪ еңЁеҮҪж•°з»„д»¶дёӯжү§иЎҢеүҜдҪңз”Ёж“ҚдҪң

import React, { useState, useEffect } from 'react';

function Example() {
  const [count, setCount] = useState(0);

  // Similar to componentDidMount and componentDidUpdate:  
  useEffect(() => {    
    // Update the document title using the browser API    
    document.title = `You clicked ${count} times`;  
  });

  return (
    <div>
      <p>You clicked {count} times</p>
      <button onClick={() => setCount(count + 1)}>
        Click me
      </button>
    </div>
  );
}

иҝҷж®өд»Јз ҒеҹәдәҺдёҠдёҖз« иҠӮдёӯзҡ„и®Ўж•°еҷЁзӨәдҫӢиҝӣиЎҢдҝ®ж”№пјҢжҲ‘д»¬дёәи®Ўж•°еҷЁеўһеҠ дәҶдёҖдёӘе°ҸеҠҹиғҪпјҡе°Ҷ document зҡ„ title и®ҫзҪ®дёәеҢ…еҗ«дәҶзӮ№еҮ»ж¬Ўж•°зҡ„ж¶ҲжҒҜгҖӮ

ж•°жҚ®иҺ·еҸ–пјҢи®ҫзҪ®и®ўйҳ…д»ҘеҸҠжүӢеҠЁжӣҙж”№ React з»„д»¶дёӯзҡ„ DOM йғҪеұһдәҺеүҜдҪңз”ЁгҖӮдёҚз®ЎдҪ зҹҘдёҚзҹҘйҒ“иҝҷдәӣж“ҚдҪңпјҢжҲ–жҳҜвҖңеүҜдҪңз”ЁвҖқиҝҷдёӘеҗҚеӯ—пјҢеә”иҜҘйғҪеңЁз»„д»¶дёӯдҪҝз”ЁиҝҮе®ғд»¬гҖӮ

жҸҗзӨә

еҰӮжһңдҪ зҶҹжӮү React class зҡ„з”ҹе‘Ҫе‘ЁжңҹеҮҪж•°пјҢдҪ еҸҜд»ҘжҠҠ useEffect Hook зңӢеҒҡ componentDidMountпјҢcomponentDidUpdate е’Ң componentWillUnmount иҝҷдёүдёӘеҮҪж•°зҡ„з»„еҗҲгҖӮ

еңЁ React з»„д»¶дёӯжңүдёӨз§Қеёёи§ҒеүҜдҪңз”Ёж“ҚдҪңпјҡйңҖиҰҒжё…йҷӨзҡ„е’ҢдёҚйңҖиҰҒжё…йҷӨзҡ„гҖӮжҲ‘д»¬жқҘжӣҙд»”з»Ҷең°зңӢдёҖдёӢд»–д»¬д№Ӣй—ҙзҡ„еҢәеҲ«гҖӮ

ж— йңҖжё…йҷӨзҡ„ effect¶

жңүж—¶еҖҷпјҢжҲ‘д»¬еҸӘжғі**еңЁ React жӣҙж–° DOM д№ӢеҗҺиҝҗиЎҢдёҖдәӣйўқеӨ–зҡ„д»Јз ҒгҖӮ**жҜ”еҰӮеҸ‘йҖҒзҪ‘з»ңиҜ·жұӮпјҢжүӢеҠЁеҸҳжӣҙ DOMпјҢи®°еҪ•ж—Ҙеҝ—пјҢиҝҷдәӣйғҪжҳҜеёёи§Ғзҡ„ж— йңҖжё…йҷӨзҡ„ж“ҚдҪңгҖӮеӣ дёәжҲ‘д»¬еңЁжү§иЎҢе®Ңиҝҷдәӣж“ҚдҪңд№ӢеҗҺпјҢе°ұеҸҜд»ҘеҝҪз•Ҙд»–д»¬дәҶгҖӮи®©жҲ‘д»¬еҜ№жҜ”дёҖдёӢдҪҝз”Ё class е’Ң Hook йғҪжҳҜжҖҺд№Ҳе®һзҺ°иҝҷдәӣеүҜдҪңз”Ёзҡ„гҖӮ

дҪҝз”Ё class зҡ„зӨәдҫӢ¶

еңЁ React зҡ„ class з»„д»¶дёӯпјҢrender еҮҪж•°жҳҜдёҚеә”иҜҘжңүд»»дҪ•еүҜдҪңз”Ёзҡ„гҖӮдёҖиҲ¬жқҘиҜҙпјҢеңЁиҝҷйҮҢжү§иЎҢж“ҚдҪңеӨӘж—©дәҶпјҢжҲ‘д»¬еҹәжң¬дёҠйғҪеёҢжңӣеңЁ React жӣҙж–° DOM д№ӢеҗҺжүҚжү§иЎҢжҲ‘д»¬зҡ„ж“ҚдҪңгҖӮ

иҝҷе°ұжҳҜдёәд»Җд№ҲеңЁ React class дёӯпјҢжҲ‘д»¬жҠҠеүҜдҪңз”Ёж“ҚдҪңж”ҫеҲ° componentDidMount е’Ң componentDidUpdate еҮҪж•°дёӯгҖӮеӣһеҲ°зӨәдҫӢдёӯпјҢиҝҷжҳҜдёҖдёӘ React и®Ўж•°еҷЁзҡ„ class з»„д»¶гҖӮе®ғеңЁ React еҜ№ DOM иҝӣиЎҢж“ҚдҪңд№ӢеҗҺпјҢз«ӢеҚіжӣҙж–°дәҶ document зҡ„ title еұһжҖ§

class Example extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      count: 0
    };
  }

  componentDidMount() {    
    document.title = `You clicked ${this.state.count} times`;  
  }  

  componentDidUpdate() {    
    document.title = `You clicked ${this.state.count} times`;  
  }

  render() {
    return (
      <div>
        <p>You clicked {this.state.count} times</p>
        <button onClick={() => this.setState({ count: this.state.count + 1 })}>
          Click me
        </button>
      </div>
    );
  }
}

жіЁж„ҸпјҢеңЁиҝҷдёӘ class дёӯпјҢжҲ‘д»¬йңҖиҰҒеңЁдёӨдёӘз”ҹе‘Ҫе‘ЁжңҹеҮҪж•°дёӯзј–еҶҷйҮҚеӨҚзҡ„д»Јз ҒгҖӮ

иҝҷжҳҜеӣ дёәеҫҲеӨҡжғ…еҶөдёӢпјҢжҲ‘д»¬еёҢжңӣеңЁз»„д»¶еҠ иҪҪе’Ңжӣҙж–°ж—¶жү§иЎҢеҗҢж ·зҡ„ж“ҚдҪңгҖӮд»ҺжҰӮеҝөдёҠиҜҙпјҢжҲ‘д»¬еёҢжңӣе®ғеңЁжҜҸж¬ЎжёІжҹ“д№ӢеҗҺжү§иЎҢ вҖ”вҖ” дҪҶ React зҡ„ class з»„д»¶жІЎжңүжҸҗдҫӣиҝҷж ·зҡ„ж–№жі•гҖӮеҚідҪҝжҲ‘д»¬жҸҗеҸ–еҮәдёҖдёӘж–№жі•пјҢжҲ‘д»¬иҝҳжҳҜиҰҒеңЁдёӨдёӘең°ж–№и°ғз”Ёе®ғгҖӮ

зҺ°еңЁи®©жҲ‘д»¬жқҘзңӢзңӢеҰӮдҪ•дҪҝз”Ё useEffect жү§иЎҢзӣёеҗҢзҡ„ж“ҚдҪңгҖӮ

дҪҝз”Ё Hook зҡ„зӨәдҫӢ¶

жҲ‘д»¬еңЁжң¬з« иҠӮејҖе§Ӣж—¶е·Із»ҸзңӢеҲ°дәҶиҝҷдёӘзӨәдҫӢпјҢдҪҶи®©жҲ‘д»¬еҶҚд»”з»Ҷи§ӮеҜҹе®ғпјҡ

import React, { useState, useEffect } from 'react';

function Example() {
  const [count, setCount] = useState(0);

  useEffect(() => {    
    document.title = `You clicked ${count} times`;  
  });

  return (
    <div>
      <p>You clicked {count} times</p>
      <button onClick={() => setCount(count + 1)}>
        Click me
      </button>
    </div>
  );
}

useEffect еҒҡдәҶд»Җд№Ҳпјҹ йҖҡиҝҮдҪҝз”ЁиҝҷдёӘ HookпјҢдҪ еҸҜд»Ҙе‘ҠиҜү React з»„д»¶йңҖиҰҒеңЁжёІжҹ“еҗҺжү§иЎҢжҹҗдәӣж“ҚдҪңгҖӮReact дјҡдҝқеӯҳдҪ дј йҖ’зҡ„еҮҪж•°пјҲжҲ‘д»¬е°Ҷе®ғз§°д№Ӣдёә вҖңeffectвҖқпјүпјҢе№¶дё”еңЁжү§иЎҢ DOM жӣҙж–°д№ӢеҗҺи°ғз”Ёе®ғгҖӮеңЁиҝҷдёӘ effect дёӯпјҢжҲ‘д»¬и®ҫзҪ®дәҶ document зҡ„ title еұһжҖ§пјҢдёҚиҝҮжҲ‘д»¬д№ҹеҸҜд»Ҙжү§иЎҢж•°жҚ®иҺ·еҸ–жҲ–и°ғз”Ёе…¶д»–е‘Ҫд»ӨејҸзҡ„ APIгҖӮ

дёәд»Җд№ҲеңЁз»„д»¶еҶ…йғЁи°ғз”Ё useEffectпјҹ е°Ҷ useEffect ж”ҫеңЁз»„д»¶еҶ…йғЁи®©жҲ‘д»¬еҸҜд»ҘеңЁ effect дёӯзӣҙжҺҘи®ҝй—® count state еҸҳйҮҸпјҲжҲ–е…¶д»– propsпјүгҖӮжҲ‘д»¬дёҚйңҖиҰҒзү№ж®Ҡзҡ„ API жқҘиҜ»еҸ–е®ғ вҖ”вҖ” е®ғе·Із»ҸдҝқеӯҳеңЁеҮҪж•°дҪңз”ЁеҹҹдёӯгҖӮHook дҪҝз”ЁдәҶ JavaScript зҡ„й—ӯеҢ…жңәеҲ¶пјҢиҖҢдёҚз”ЁеңЁ JavaScript е·Із»ҸжҸҗдҫӣдәҶи§ЈеҶіж–№жЎҲзҡ„жғ…еҶөдёӢпјҢиҝҳеј•е…Ҙзү№е®ҡзҡ„ React APIгҖӮ

useEffect дјҡеңЁжҜҸж¬ЎжёІжҹ“еҗҺйғҪжү§иЎҢеҗ—пјҹ жҳҜзҡ„пјҢй»ҳи®Өжғ…еҶөдёӢпјҢе®ғеңЁз¬¬дёҖж¬ЎжёІжҹ“д№ӢеҗҺ*е’Ң*жҜҸж¬Ўжӣҙж–°д№ӢеҗҺйғҪдјҡжү§иЎҢгҖӮпјҲжҲ‘д»¬зЁҚеҗҺдјҡи°ҲеҲ°еҰӮдҪ•жҺ§еҲ¶е®ғгҖӮпјүдҪ еҸҜиғҪдјҡжӣҙе®№жҳ“жҺҘеҸ— effect еҸ‘з”ҹеңЁвҖңжёІжҹ“д№ӢеҗҺвҖқиҝҷз§ҚжҰӮеҝөпјҢдёҚз”ЁеҶҚеҺ»иҖғиҷ‘вҖңжҢӮиҪҪвҖқиҝҳжҳҜвҖңжӣҙж–°вҖқгҖӮReact дҝқиҜҒдәҶжҜҸж¬ЎиҝҗиЎҢ effect зҡ„еҗҢж—¶пјҢDOM йғҪе·Із»Ҹжӣҙж–°е®ҢжҜ•гҖӮ

иҜҰз»ҶиҜҙжҳҺ¶

зҺ°еңЁжҲ‘д»¬е·Із»ҸеҜ№ effect жңүдәҶеӨ§иҮҙдәҶи§ЈпјҢдёӢйқўиҝҷдәӣд»Јз Ғеә”иҜҘдёҚйҡҫзңӢжҮӮдәҶпјҡ

function Example() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    document.title = `You clicked ${count} times`;
  });
}

жҲ‘д»¬еЈ°жҳҺдәҶ count state еҸҳйҮҸпјҢе№¶е‘ҠиҜү React жҲ‘д»¬йңҖиҰҒдҪҝз”Ё effectгҖӮзҙ§жҺҘзқҖдј йҖ’еҮҪж•°з»ҷ useEffect HookгҖӮжӯӨеҮҪж•°е°ұжҳҜжҲ‘д»¬зҡ„ effectгҖӮз„¶еҗҺдҪҝз”Ё document.title жөҸи§ҲеҷЁ API и®ҫзҪ® document зҡ„ titleгҖӮжҲ‘д»¬еҸҜд»ҘеңЁ effect дёӯиҺ·еҸ–еҲ°жңҖж–°зҡ„ count еҖјпјҢеӣ дёәд»–еңЁеҮҪж•°зҡ„дҪңз”ЁеҹҹеҶ…гҖӮеҪ“ React жёІжҹ“з»„д»¶ж—¶пјҢдјҡдҝқеӯҳе·ІдҪҝз”Ёзҡ„ effectпјҢе№¶еңЁжӣҙж–°е®Ң DOM еҗҺжү§иЎҢе®ғгҖӮиҝҷдёӘиҝҮзЁӢеңЁжҜҸж¬ЎжёІжҹ“ж—¶йғҪдјҡеҸ‘з”ҹпјҢеҢ…жӢ¬йҰ–ж¬ЎжёІжҹ“гҖӮ

з»ҸйӘҢдё°еҜҢзҡ„ JavaScript ејҖеҸ‘дәәе‘ҳеҸҜиғҪдјҡжіЁж„ҸеҲ°пјҢдј йҖ’з»ҷ useEffect зҡ„еҮҪж•°еңЁжҜҸж¬ЎжёІжҹ“дёӯйғҪдјҡжңүжүҖдёҚеҗҢпјҢиҝҷжҳҜеҲ»ж„Ҹдёәд№Ӣзҡ„гҖӮдәӢе®һдёҠиҝҷжӯЈжҳҜжҲ‘д»¬еҸҜд»ҘеңЁ effect дёӯиҺ·еҸ–жңҖж–°зҡ„ count зҡ„еҖјпјҢиҖҢдёҚз”ЁжӢ…еҝғе…¶иҝҮжңҹзҡ„еҺҹеӣ гҖӮжҜҸж¬ЎжҲ‘д»¬йҮҚж–°жёІжҹ“пјҢйғҪдјҡз”ҹжҲҗ*ж–°зҡ„* effectпјҢжӣҝжҚўжҺүд№ӢеүҚзҡ„гҖӮжҹҗз§Қж„Ҹд№үдёҠи®ІпјҢeffect жӣҙеғҸжҳҜжёІжҹ“з»“жһңзҡ„дёҖйғЁеҲҶ вҖ”вҖ” жҜҸдёӘ effect вҖңеұһдәҺвҖқдёҖж¬Ўзү№е®ҡзҡ„жёІжҹ“гҖӮжҲ‘д»¬е°ҶеңЁжң¬з« иҠӮеҗҺз»ӯйғЁеҲҶжӣҙжё…жҘҡең°дәҶи§Јиҝҷж ·еҒҡзҡ„ж„Ҹд№үгҖӮ

жҸҗзӨә

дёҺ componentDidMount жҲ– componentDidUpdate дёҚеҗҢпјҢдҪҝз”Ё useEffect и°ғеәҰзҡ„ effect дёҚдјҡйҳ»еЎһжөҸи§ҲеҷЁжӣҙж–°еұҸе№•пјҢиҝҷи®©дҪ зҡ„еә”з”ЁзңӢиө·жқҘе“Қеә”жӣҙеҝ«гҖӮеӨ§еӨҡж•°жғ…еҶөдёӢпјҢeffect дёҚйңҖиҰҒеҗҢжӯҘең°жү§иЎҢгҖӮеңЁдёӘеҲ«жғ…еҶөдёӢпјҲдҫӢеҰӮжөӢйҮҸеёғеұҖпјүпјҢжңүеҚ•зӢ¬зҡ„ useLayoutEffect Hook дҫӣдҪ дҪҝз”ЁпјҢе…¶ API дёҺ useEffect зӣёеҗҢгҖӮ

йңҖиҰҒжё…йҷӨзҡ„ effect¶

д№ӢеүҚпјҢжҲ‘д»¬з ”з©¶дәҶеҰӮдҪ•дҪҝз”ЁдёҚйңҖиҰҒжё…йҷӨзҡ„еүҜдҪңз”ЁпјҢиҝҳжңүдёҖдәӣеүҜдҪңз”ЁжҳҜйңҖиҰҒжё…йҷӨзҡ„гҖӮдҫӢеҰӮ**и®ўйҳ…еӨ–йғЁж•°жҚ®жәҗ**гҖӮиҝҷз§Қжғ…еҶөдёӢпјҢжё…йҷӨе·ҘдҪңжҳҜйқһеёёйҮҚиҰҒзҡ„пјҢеҸҜд»ҘйҳІжӯўеј•иө·еҶ…еӯҳжі„йңІпјҒзҺ°еңЁи®©жҲ‘д»¬жқҘжҜ”иҫғдёҖдёӢеҰӮдҪ•з”Ё Class е’Ң Hook жқҘе®һзҺ°гҖӮ

дҪҝз”Ё Class зҡ„зӨәдҫӢ¶

еңЁ React class дёӯпјҢдҪ йҖҡеёёдјҡеңЁ componentDidMount дёӯи®ҫзҪ®и®ўйҳ…пјҢе№¶еңЁ componentWillUnmount дёӯжё…йҷӨе®ғгҖӮдҫӢеҰӮпјҢеҒҮи®ҫжҲ‘д»¬жңүдёҖдёӘ ChatAPI жЁЎеқ—пјҢе®ғе…Ғи®ёжҲ‘д»¬и®ўйҳ…еҘҪеҸӢзҡ„еңЁзәҝзҠ¶жҖҒгҖӮд»ҘдёӢжҳҜжҲ‘д»¬еҰӮдҪ•дҪҝз”Ё class и®ўйҳ…е’ҢжҳҫзӨәиҜҘзҠ¶жҖҒпјҡ

class FriendStatus extends React.Component {
  constructor(props) {
    super(props);
    this.state = { isOnline: null };
    this.handleStatusChange = this.handleStatusChange.bind(this);
  }

  componentDidMount() {    
    ChatAPI.subscribeToFriendStatus(      
      this.props.friend.id,      
      this.handleStatusChange    
    );  
  }  

  componentWillUnmount() {    
    ChatAPI.unsubscribeFromFriendStatus(      
      this.props.friend.id,      
      this.handleStatusChange    
    );  
  }  

  handleStatusChange(status) {    
    this.setState({      
      isOnline: status.isOnline    
    });  
  }

  render() {
    if (this.state.isOnline === null) {
      return 'Loading...';
    }
    return this.state.isOnline ? 'Online' : 'Offline';
  }
}

дҪ дјҡжіЁж„ҸеҲ° componentDidMount е’Ң componentWillUnmount д№Ӣй—ҙзӣёдә’еҜ№еә”гҖӮдҪҝз”Ёз”ҹе‘Ҫе‘ЁжңҹеҮҪж•°иҝ«дҪҝжҲ‘д»¬жӢҶеҲҶиҝҷдәӣйҖ»иҫ‘д»Јз ҒпјҢеҚідҪҝиҝҷдёӨйғЁеҲҶд»Јз ҒйғҪдҪңз”ЁдәҺзӣёеҗҢзҡ„еүҜдҪңз”ЁгҖӮ

жіЁж„Ҹ

зңје°–зҡ„иҜ»иҖ…еҸҜиғҪе·Із»ҸжіЁж„ҸеҲ°дәҶпјҢиҝҷдёӘзӨәдҫӢиҝҳйңҖиҰҒзј–еҶҷ componentDidUpdate ж–№жі•жүҚиғҪдҝқиҜҒе®Ңе…ЁжӯЈзЎ®гҖӮжҲ‘д»¬е…ҲжҡӮж—¶еҝҪз•ҘиҝҷдёҖзӮ№пјҢжң¬з« иҠӮдёӯеҗҺз»ӯйғЁеҲҶдјҡд»Ӣз»Қе®ғгҖӮ

дҪҝз”Ё Hook зҡ„зӨәдҫӢ¶

еҰӮдҪ•дҪҝз”Ё Hook зј–еҶҷиҝҷдёӘз»„д»¶гҖӮ

дҪ еҸҜиғҪи®ӨдёәйңҖиҰҒеҚ•зӢ¬зҡ„ effect жқҘжү§иЎҢжё…йҷӨж“ҚдҪңгҖӮдҪҶз”ұдәҺж·»еҠ е’ҢеҲ йҷӨи®ўйҳ…зҡ„д»Јз Ғзҡ„зҙ§еҜҶжҖ§пјҢжүҖд»Ҙ useEffect зҡ„и®ҫи®ЎжҳҜеңЁеҗҢдёҖдёӘең°ж–№жү§иЎҢгҖӮеҰӮжһңдҪ зҡ„ effect иҝ”еӣһдёҖдёӘеҮҪж•°пјҢReact е°ҶдјҡеңЁжү§иЎҢжё…йҷӨж“ҚдҪңж—¶и°ғз”Ёе®ғпјҡ

import React, { useState, useEffect } from 'react';

function FriendStatus(props) {
  const [isOnline, setIsOnline] = useState(null);

  useEffect(() => {    
    function handleStatusChange(status) {      
      setIsOnline(status.isOnline);    
    }
    ChatAPI.subscribeToFriendStatus(props.friend.id, handleStatusChange);   
    // Specify how to clean up after this effect:    
    return function cleanup() {      
      ChatAPI.unsubscribeFromFriendStatus(props.friend.id, handleStatusChange);    
    };  
  });

  if (isOnline === null) {
    return 'Loading...';
  }
  return isOnline ? 'Online' : 'Offline';

жңӘжқҘзүҲжң¬пјҢеҸҜиғҪдјҡеңЁжһ„е»әж—¶иҮӘеҠЁж·»еҠ з¬¬дәҢдёӘеҸӮж•°гҖӮ

жіЁж„Ҹпјҡ

еҰӮжһңдҪ иҰҒдҪҝз”ЁжӯӨдјҳеҢ–ж–№ејҸпјҢиҜ·зЎ®дҝқж•°з»„дёӯеҢ…еҗ«дәҶ**жүҖжңүеӨ–йғЁдҪңз”ЁеҹҹдёӯдјҡйҡҸж—¶й—ҙеҸҳеҢ–е№¶дё”еңЁ effect дёӯдҪҝз”Ёзҡ„еҸҳйҮҸ**пјҢеҗҰеҲҷдҪ зҡ„д»Јз Ғдјҡеј•з”ЁеҲ°е…ҲеүҚжёІжҹ“дёӯзҡ„ж—§еҸҳйҮҸгҖӮеҸӮйҳ…ж–ҮжЎЈпјҢдәҶи§ЈжӣҙеӨҡе…ідәҺеҰӮдҪ•еӨ„зҗҶеҮҪж•°д»ҘеҸҠж•°з»„йў‘з№ҒеҸҳеҢ–ж—¶зҡ„жҺӘж–ҪеҶ…е®№гҖӮ

еҰӮжһңжғіжү§иЎҢеҸӘиҝҗиЎҢдёҖж¬Ўзҡ„ effectпјҲд»…еңЁз»„д»¶жҢӮиҪҪе’ҢеҚёиҪҪж—¶жү§иЎҢпјүпјҢеҸҜд»Ҙдј йҖ’дёҖдёӘз©әж•°з»„пјҲ[]пјүдҪңдёәз¬¬дәҢдёӘеҸӮж•°гҖӮиҝҷе°ұе‘ҠиҜү React дҪ зҡ„ effect дёҚдҫқиө–дәҺ props жҲ– state дёӯзҡ„д»»дҪ•еҖјпјҢжүҖд»Ҙе®ғж°ёиҝңйғҪдёҚйңҖиҰҒйҮҚеӨҚжү§иЎҢгҖӮиҝҷе№¶дёҚеұһдәҺзү№ж®Ҡжғ…еҶө вҖ”вҖ” е®ғдҫқз„¶йҒөеҫӘдҫқиө–ж•°з»„зҡ„е·ҘдҪңж–№ејҸгҖӮ

еҰӮжһңдҪ дј е…ҘдәҶдёҖдёӘз©әж•°з»„пјҲ[]пјүпјҢeffect еҶ…йғЁзҡ„ props е’Ң state е°ұдјҡдёҖзӣҙжӢҘжңүе…¶еҲқе§ӢеҖјгҖӮе°Ҫз®Ўдј е…Ҙ [] дҪңдёәз¬¬дәҢдёӘеҸӮж•°жӣҙжҺҘиҝ‘еӨ§е®¶жӣҙзҶҹжӮүзҡ„ componentDidMount е’Ң componentWillUnmount жҖқз»ҙжЁЎејҸпјҢдҪҶжҲ‘д»¬жңүжӣҙеҘҪзҡ„ж–№ејҸжқҘйҒҝе…ҚиҝҮдәҺйў‘з№Ғзҡ„йҮҚеӨҚи°ғз”Ё effectгҖӮйҷӨжӯӨд№ӢеӨ–пјҢиҜ·и®°еҫ— React дјҡзӯүеҫ…жөҸи§ҲеҷЁе®ҢжҲҗз”»йқўжёІжҹ“д№ӢеҗҺжүҚдјҡе»¶иҝҹи°ғз”Ё useEffectпјҢеӣ жӯӨдјҡдҪҝеҫ—йўқеӨ–ж“ҚдҪңеҫҲж–№дҫҝгҖӮ

жҲ‘д»¬жҺЁиҚҗеҗҜз”Ё eslint-plugin-react-hooks дёӯзҡ„ exhaustive-deps и§„еҲҷгҖӮжӯӨи§„еҲҷдјҡеңЁж·»еҠ й”ҷиҜҜдҫқиө–ж—¶еҸ‘еҮәиӯҰе‘Ҡе№¶з»ҷеҮәдҝ®еӨҚе»әи®®гҖӮ

дёӢдёҖжӯҘ¶

жҒӯе–ңдҪ пјҒе®ҢжҲҗжң¬з« иҠӮеӯҰд№ пјҢеёҢжңӣе…ідәҺ effect зҡ„еӨ§еӨҡж•°й—®йўҳйғҪеҫ—еҲ°дәҶи§Јзӯ”гҖӮдҪ е·Із»ҸеӯҰд№ дәҶ State Hook е’Ң Effect HookпјҢе°Ҷе®ғд»¬з»“еҗҲиө·жқҘдҪ еҸҜд»ҘеҒҡеҫҲеӨҡдәӢжғ…дәҶгҖӮе®ғд»¬ж¶өзӣ–дәҶеӨ§еӨҡж•°дҪҝз”Ё class зҡ„з”ЁдҫӢ вҖ”вҖ” еҰӮжһңжІЎжңүпјҢдҪ еҸҜд»ҘжҹҘзңӢе…¶д»–зҡ„ HookгҖӮ

жҲ‘д»¬зңӢеҲ°дәҶ Hook еҰӮдҪ•и§ЈеҶіз®Җд»Ӣз« иҠӮдёӯеҠЁжңәйғЁеҲҶжҸҗеҮәзҡ„й—®йўҳгҖӮжҲ‘д»¬д№ҹеҸ‘зҺ° effect зҡ„жё…йҷӨжңәеҲ¶еҸҜд»ҘйҒҝе…Қ componentDidUpdate е’Ң componentWillUnmount дёӯзҡ„йҮҚеӨҚпјҢеҗҢж—¶и®©зӣёе…ізҡ„д»Јз Ғе…іиҒ”жӣҙеҠ зҙ§еҜҶпјҢеё®еҠ©жҲ‘д»¬йҒҝе…ҚдёҖдәӣ bugгҖӮжҲ‘д»¬иҝҳзңӢеҲ°дәҶжҲ‘д»¬еҰӮдҪ•ж №жҚ® effect зҡ„еҠҹиғҪеҲҶйҡ”д»–д»¬пјҢиҝҷжҳҜеңЁ class дёӯж— жі•еҒҡеҲ°зҡ„гҖӮ

жӯӨж—¶дҪ еҸҜиғҪдјҡеҘҪеҘҮ Hook жҳҜеҰӮдҪ•е·ҘдҪңзҡ„гҖӮеңЁдёӨж¬ЎжёІжҹ“й—ҙпјҢReactеҰӮдҪ•зҹҘйҒ“е“ӘдёӘ useState и°ғз”ЁеҜ№еә”дәҺе“ӘдёӘ state еҸҳйҮҸпјҹReact еҸҲжҳҜеҰӮдҪ•еҢ№й…ҚеүҚеҗҺдёӨж¬ЎжёІжҹ“дёӯзҡ„жҜҸдёҖдёӘ effect зҡ„пјҹ
