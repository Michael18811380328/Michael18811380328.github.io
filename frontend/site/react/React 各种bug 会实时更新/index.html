export default class StringUtils {

    isEmpty(str) {
        let flag = false;
        if (!str || str && this.isNull(str) || !this.removeAllSpaces(str)) {
            flag = true;
        }
        return flag;
    }

    //ίΙνόΨφόαψίΡοόεΚύσ║
    isNull(str) {
        if (str === "") return true;
        let regu = "^[ ]+$";
        let re = new RegExp(regu);
        return re.test(str);
    }

    //ϋΟ╖ίΠΨϊ╕φόΨΘύΛ╢όΑΒϊ╕ΜϋΟ╖ίΠΨίΡΟϊ╕νϊ╜ΞίΡοίΙβίΚΞϊ╕νϊ╜Ξ
    getNameSubstr(str) {
        if (str) {

            if (!this.isChinese(str)) {
                if (str.length >= 2) {
                    return str.substr(str.length - 2, 2)
                } else {
                    return str.substr(str.length - 1, 1)
                }
            } else {
                if (str.length >= 2) {
                    return str.substr(0, 2)
                } else {
                    return str.substr(0, 1)
                }
            }
        } else {
            return '';
        }
    }

    //ίΙνόΨφόαψίΡοϊ╕φόΨΘίΤΝϋΜ▒όΨΘ
    isChinese(str) {
        if (/.*[\u4e00-\u9fa5]+.*$/.test(str)) {
            return false;
        }
        return true;
    }

    /***
     * ίΟ╗ώβνόΚΑόεΚύσ║όι╝
     * @param str
     * @returns {string}
     */
    removeAllSpaces(str) {
        try {
            return str.replace(/\s+/g, "")
        } catch (e) {
            return ''
        }
    }

    /***
     * ίΟ╗ώβνϊ╕νίν┤ύσ║όι╝
     * @param str
     * @returns {string}
     */
    removeBothSpaces(str) {
        try {
            return str.replace(/^\s+|\s+$/g, "")
        } catch (e) {
            return ''
        }
    }
}
