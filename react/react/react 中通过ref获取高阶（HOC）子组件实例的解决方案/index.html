    react дёӯйҖҡиҝҮrefиҺ·еҸ–й«ҳйҳ¶пјҲHOCпјүеӯҗз»„д»¶е®һдҫӢзҡ„и§ЈеҶіж–№жЎҲ¶

    д»ҠеӨ©еҶҷreactйЎ№зӣ®йҒҮеҲ°дёҖдёӘзҲ¶еӯҗз»„д»¶йҖҡдҝЎзҡ„й—®йўҳгҖӮиҝҷжҳҜдёҖдёӘйқһеёёеёёи§„зҡ„й—®йўҳдәҶпјҢйҡҸдҫҝжҗңдёҖдёӢе°ұиғҪеҫ—еҲ°и§ЈеҶіж–№жЎҲгҖӮжҖ»дҪ“жқҘиҜҙеҸҜд»ҘеҲҶдёәдёӨзұ»пјҡ

    1. еӯҗз»„д»¶йңҖиҰҒиҺ·еҸ–зҲ¶з»„д»¶зҡ„дҝЎжҒҜпјҢиҝҷйҖҡиҝҮpropsе°ұеҸҜд»Ҙи§ЈеҶіпјӣ
    2. зҲ¶з»„д»¶йңҖиҰҒзҹҘйҒ“еӯҗз»„д»¶зҡ„дҝЎжҒҜпјҢиҝҷеҸҜд»ҘйҖҡиҝҮrefи§ЈеҶігҖӮ

    жҲ‘д»¬иҝҷйҮҢи®Ізҡ„еұһдәҺеҗҺиҖ…пјҢдҪҶжҳҜеҸҲжңүдәӣзү№ж®ҠпјҢзү№ж®Ҡе°ұеңЁдәҺеӯҗз»„д»¶жҳҜдёӘй«ҳйҳ¶з»„д»¶пјҢжҜ”еҰӮдҪҝз”Ё@connect @withRouterеҢ…иЈ№иҝҮзҡ„з»„д»¶пјҲе…¶е®һеӨ§йғЁеҲҶз»„д»¶йғҪдјҡиў«иҝҷдёӨдёӘеҢ…иЈ№пјүпјҢе…·дҪ“зӨәдҫӢеҰӮдёӢпјҡ

    @withRouter
export default class childComponent extends Component {
  constructor(props) {
    super(props);
    this.state = {};
  }
  render() {
    return (<div>this is childComponent</div>)
  }
}

@withRouter
export default class parentComponent extends Component {
  constructor(props) {

    this.state = {}; 
  }
  render () {
    return <childComponent ref={(v) => { this.childCp = v; }}/>
  }
 }

    дёҠйқўзҡ„childComponentиў«withRouterеҢ…иЈ№иҝҮдёҖйҒҚеҗҺпјҢиҝҷж—¶еҖҷдҪ еңЁparentComponentдёӯйҖҡиҝҮrefиҺ·еҸ–еҲ°зҡ„жҳҜе№¶дёҚдјҡжҳҜchildComponentпјҢиҖҢжҳҜwithRouterз»„д»¶гҖӮиҝҷе°ұжҜ”иҫғе°ҙе°¬дәҶпјҢжҲ‘д»¬еӨ§еӨҡж•°жғ…еҶөиӮҜе®ҡжҳҜйңҖиҰҒиҺ·еҸ–иҮӘе·ұеҶҷзҡ„з»„д»¶е®һдҫӢзҡ„гҖӮжңүдёҖзӮ№йңҖиҰҒи®ІжҳҺзҷҪпјҡе°ұжҳҜйҖҡиҝҮrefиҺ·еҸ–еҲ°зҡ„дёҚжҳҜchildComponentпјҢеңЁеҺҹзҗҶдёҠжҳҜеҜ№зҡ„пјҢеҰӮжһңиҺ·еҸ–еҲ°жҳҜchildComponentз»„д»¶йӮЈжүҚжҳҜжңүй—®йўҳзҡ„пјҢжңүжӮ–дјҰзҗҶзҹҘйҒ“е“ҮгҖӮ

    ж—ўз„¶йҖҡиҝҮе®ҳж–№жҸҗдҫӣзҡ„refж— жі•иҺ·еҸ–еҲ°жҲ‘д»¬жғіиҰҒзҡ„refпјҢйӮЈжҲ‘д»¬е°ұжқҘд»”з»ҶжғідёӢrefиҺ·еҸ–еҲ°зҡ„жҳҜе•ҘпјҹжҲ‘д»¬еӣһеҪ’еҲ°javascriptиҜӯиЁҖеұӮйқўжқҘзңӢпјҢйӮЈдёҚе°ұжҳҜз»„д»¶дёӯзҡ„thisд№ҲгҖӮ

    @withRouter
export default class childComponent extends Component {

    super(props);
    this.state = {}; // жҲ‘д»¬йңҖиҰҒиҺ·еҸ–еҲ°зҡ„е°ұжҳҜиҝҷдёӘthisиҖҢе·І
  }
  render() {
    return (<div>this is childComponent</div>)
  }
}

    зҹҘйҒ“жҲ‘д»¬йңҖиҰҒиҺ·еҸ–еҲ°зҡ„жҳҜе•ҘдәҶпјҢйӮЈе°ұеҘҪеҠһдәҶпјҢжҲ‘з»ҷchildComponentдј дёҖдёӘpropдё“й—ЁжқҘgetиҝҷдёӘthisдёҚе°ұеҘҪдәҶпјҢжҜ”еҰӮдҪҝз”ЁgetInstanceпјҡ

    perfect ! й—®йўҳи§ЈеҶідәҶпјҢиҝҷж ·жҲ‘дёҚз®ЎдҪ жҖҺд№Ҳз”Ёе•Ҙй«ҳйҳ¶з»„д»¶гҖҒз”ЁеӨҡе°‘дёӘй«ҳйҳ¶з»„д»¶еҢ…иЈ№жҲ‘д»¬childComponentпјҢжҲ‘д»¬йғҪеҸҜд»ҘйҖҡиҝҮдёҖдёӘgetInstanceпјҢз©ҝи¶ҠеҚғеұұдёҮж°ҙзӣҙжҺҘиҺ·еҸ–childComponentе®һдҫӢгҖӮ

    еҪ“з„¶е®ҢзҫҺд№ҹжҳҜзӣёеҜ№зҡ„пјҢжҜ”еҰӮдёҠйқўзҡ„ж–№жЎҲдёӯпјҢжҲ‘д»¬еҫ—еңЁжҜҸдёҖдёӘchildComponentзҡ„жһ„йҖ еҮҪж•°дёӯеҶҷйӮЈж®өжҡҙйңІthisзҡ„д»Јз ҒпјҢйә»зғҰгҖҒиҙ№еҠІгҖӮиҝҷж—¶еҖҷжҲ‘д»¬еҸҜд»ҘеҶҷдёҖдёӘHOCдё“й—ЁжқҘеҒҡиҝҷд»¶дәӢжғ…пјҢжҜ”еҰӮwithRef:

    // еҸӘеҒҡдёҖд»¶дәӢпјҢжҠҠ`WrappedComponent`еӣһдј дёӘ`getInstance`пјҲеҰӮжһңжңүзҡ„иҜқпјү
export default (WrappedComponent) => {
  return class withRef extends Component {
    static displayName = `withRef(${WrappedComponent.displayName || WrappedComponent.name || 'Component'})`;
    render() {
      // иҝҷйҮҢйҮҚж–°е®ҡд№үдёҖдёӘpropsзҡ„еҺҹеӣ жҳҜ:
      // дҪ зӣҙжҺҘеҺ»дҝ®ж”№this.props.refеңЁreactејҖеҸ‘жЁЎејҸдёӢдјҡжҠҘй”ҷпјҢдёҚе…Ғи®ёдҪ еҺ»дҝ®ж”№
      const props = {
        ...this.props,
      };
      // еңЁиҝҷйҮҢжҠҠgetInstanceиөӢеҖјз»ҷrefпјҢ
      // дј з»ҷ`WrappedComponent`пјҢиҝҷж ·е°ұgetInstanceиғҪиҺ·еҸ–еҲ°`WrappedComponent`е®һдҫӢ
      // ж„ҹи°ўиҜ„и®әеҢәзҡ„[yangshenghaha]еҗҢеӯҰзҡ„е®Ңе–„
      props.ref = (el)=>{
          this.props.getInstance && this.props.getInstance(el);this.props.ref && this.props.ref(el);
      }
      return (
        <WrappedComponent {...props} />
      );
    }
  };
};
