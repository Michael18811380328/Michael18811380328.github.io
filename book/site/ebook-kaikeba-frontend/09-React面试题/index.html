function useClock() {
  const [date, setDate] = useState(new Date());
  useEffect(() => {
    // didMount йҳ¶ж®өжү§иЎҢ
    const timer = setInterval(() => {
      setDate(newDate());
    }, 1000);
    // UNmountйҳ¶ж®өпјҢжё…йҷӨе®ҡж—¶еҷЁ
    return () => clearInterval(timer);
  }, []);
  return date;
}
