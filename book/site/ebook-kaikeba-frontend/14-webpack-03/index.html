    жң¬ең°ејҖеҸ‘жңҚеҠЎ¶

    еүҚз«Ҝи·Ёеҹҹи®ҫзҪ®д»ЈзҗҶ proxy¶

    й—®йўҳпјҡеүҚеҗҺз«ҜеҲҶзҰ»ејҖеҸ‘ж—¶пјҢеҗҺз«ҜиҝҳжІЎжңүеҶҷеҘҪд»Јз ҒпјҢеүҚз«ҜеҸӘиғҪиҮӘе·ұжҗӯе»әдёҖдёӘдёҙж—¶жңҚеҠЎеҷЁгҖӮеҰӮжһңдёҙж—¶зҡ„жңҚеҠЎеҷЁе’Ң webpack-dev-server зҡ„з«ҜеҸЈдёҚдёҖиҮҙпјҢйӮЈд№ҲдјҡеҮәзҺ°и·Ёеҹҹзҡ„й—®йўҳпјҢжүҖд»ҘйңҖиҰҒи§ЈеҶіпјҢеҸҜд»Ҙи®ҫзҪ®д»ЈзҗҶгҖӮ

    еҗҺз«Ҝдјҡз»ҷеҮәжҺҘеҸЈж–ҮжЎЈе’ҢиҒ”и°ғж—ҘжңҹпјӣеүҚз«ҜеңЁиҒ”и°ғеүҚпјҢйңҖиҰҒзӢ¬з«ӢејҖеҸ‘пјӣдёҠзәҝеҗҺдҪҝз”Ё nginx еӨ„зҗҶи·ЁеҹҹгҖӮ

    еҸӘиҰҒ UI з»ҷеҮәдәӨзЁҝж—ҘжңҹпјҢеҜ№жҺҘе®ҢжҲҗпјҢеүҚз«Ҝе°ұеҸҜд»ҘејҖе§Ӣе·ҘдҪңгҖӮ

    // дҪҝз”Ё express mock ж•°жҚ®пјҢжңҚеҠЎеҷЁз«ҜеҸЈеҸ· 8080 
const express = require('express');
const app = express();

app.get("api/comment", (req, res) => {
  res.json({
    data: 'hi'
  })
})

app.listen("8080")
// з„¶еҗҺеңЁз»Ҳз«ҜдёӯзӣҙжҺҘ node жү§иЎҢиҝҷдёӘж–Үд»¶пјҢеҚіеҸҜејҖеҗҜжң¬ең°дёҙж—¶жңҚеҠЎеҷЁ

    йӮЈд№ҲжҲ‘д»¬еҸҜд»ҘеңЁ webpack и®ҫзҪ®дёӯжӣҙж”№ proxy еұһжҖ§пјҡйҒҮеҲ° /api ејҖеӨҙзҡ„иҜ·жұӮе°ұиҪ¬еҸ‘еҲ° 8080 з«ҜеҸЈ

    {
  devServer: {
    contentBase: "./dist",
    open: true,
    proxy: {
      "/api": {
        target: "http://localhost:8080",
      }
    },
    port: 8000,
  }
}

    HMR зғӯжЁЎеқ—жӣҝжҚў¶

    hot module replacement йҰ–е…ҲеңЁжөҸи§ҲеҷЁдёӯзӮ№еҮ»дәӨдә’пјҢжӣҙж”№д»Јз ҒеҗҺпјҢжөҸи§ҲеҷЁеҲ·ж–°пјҢдҪҶжҳҜе·Іжңүзҡ„дәӨдә’дёҚж¶ҲеӨұпјҢзӣҙжҺҘж”№еҸҳйўңиүІжҲ–иҖ…еҶ…е®№пјӣеңЁи°ғиҠӮж ·ејҸж—¶еҫҲж–№дҫҝгҖӮ

    й…ҚзҪ®ж–Үд»¶еӨҙйғЁеј•е…Ҙ webpackпјҢз„¶еҗҺжӣҙж”№й…ҚзҪ®пјҡ

    const webpack = require('webpack');

{
  devServer: {
    contentBase: "./dist",
    open: true,
    hot: true,
    hotOnly: true, // ејҖеҗҜ HMR
  },
  plugin: [
    new CleanWebpackPlugin(),
    new HtmlWebpackPlugin({
      template: "src/index.html"
    }),
    new webpack.HotModuleReplacementPlugin(), // ејҖеҗҜ HMR
  ],
}

    CSS зҡ„ HMR¶

    MiniCssExtractPlugin еңЁејҖеҸ‘йҳ¶ж®өпјҢеҸҜиғҪе’Ң webpackDevServer зҡ„ hot зғӯеҗҜеҠЁеҶІзӘҒгҖӮеҪ“жӣҙж”№дәҶж ·ејҸеҗҺпјҢдёҚдјҡиҮӘеҠЁеҲ·ж–°з•Ңйқўдёӯзҡ„йўңиүІгҖӮжӯӨж—¶жҠҠиҝҷдёӘ mini еҺ»жҺүпјҢж”№жҲҗй»ҳи®Өзҡ„ style-loader еҚіеҸҜе®һзҺ°гҖӮ

    жіЁж„Ҹпјҡ

    • HMRзғӯжЁЎеқ—жӣҙж–°з”ЁдәҺејҖеҸ‘зҺҜеўғпјҢжҠҪеҸ–CSSжҸ’д»¶з”ЁдәҺз”ҹдә§зҺҜеўғпјҢи®ҫзҪ®дёҚеҗҢзҡ„ webpack й…ҚзҪ®гҖӮ

    • HMR дёҚж”ҜжҢҒ chunkHash contentHash пјҲchunkHash з”ЁдәҺ JS жЁЎеқ—еҗҚз§°пјҢcontentHash з”ЁдәҺ css жЁЎеқ—еҗҚз§°пјү

    JS зҡ„ HMR¶

    еҺҹзҗҶпјҡйҖҡиҝҮ module.hot.accept жқҘи§ӮеҜҹжЁЎеқ—жӣҙж–°пјҢз„¶еҗҺзӣҙжҺҘжӣҙж–°иҠӮзӮ№пјҲдёӢйқўд»Јз ҒжҳҜеҺҹзҗҶпјү

    import number from "./number";
import comment from "./comment";

// еҜје…ҘдёӨдёӘжЁЎеқ—е№¶еҲқе§ӢеҢ–
number();
comment();

// еҰӮжһңж”ҜжҢҒжЁЎеқ—жӣҙж–°
if (module.hot) {
  // еҰӮжһңи§ӮеҜҹеҲ°жЁЎеқ—жӣҙж–°
  module.hot.accept('./b', function() {
    document.body.removeChild(document.getElementById("number"));
    number();
  });
}

    pollyfill¶

    й—®йўҳпјҡBabel еҸҜд»ҘиҪ¬жҚўз®ӯеӨҙеҮҪж•°пјҢдҪҶж— жі•иҪ¬жҚў PromiseпјҢйңҖиҰҒдҪҝз”Ёеһ«зүҮ polyfill пјҲES6+зҡ„и§„иҢғеә“пјү

}; };
