Day1 webpackй…ҚзҪ®жҰӮеҝө¶

2021-10-25 д»ҘиҖғд»Јз»ғпјҡ

  • жҰӮиҝ° Webpack дё»иҰҒд»Җд№Ҳз”ЁйҖ”пјҢжҖҺд№Ҳе®үиЈ…пјҢchunk bundle жЁЎеқ—зҡ„жҰӮеҝөпјҹ
  • ж ёеҝғзҡ„й…ҚзҪ®е“ӘдәӣпјҹеӣӣдёӘпјӣеҚ•е…ҘеҸЈе’ҢеӨҡе…ҘеҸЈжҖҺд№Ҳй…ҚзҪ®пјҹ
  • loader жҳҜд»Җд№Ҳз”ЁйҖ”пјҢеёёз”Ёзҡ„дә”дёӘloaderжҳҜд»Җд№ҲпјҢе…·дҪ“жҖҺд№ҲдҪҝз”Ё
  • plugin жҳҜд»Җд№Ҳз”ЁйҖ”пјҢеёёз”Ёзҡ„дёӨдёӘжҳҜд»Җд№Ҳпјҹ

1.webpackз®Җд»Ӣ¶

webpack is a module bundler (жЁЎеқ—жү“еҢ…вјҜе·Ҙе…·)

WebpackжҳҜдёҖдёӘжү“еҢ…жЁЎеқ—еҢ– JavaScript зҡ„е·Ҙе…·пјҢе®ғдјҡд»Һе…ҘеҸЈжЁЎеқ—еҮәеҸ‘пјҢиҜҶеҲ«еҮәжәҗз Ғдёӯзҡ„жЁЎеқ—еҢ–еҜје…ҘиҜӯпӨҶпјҢйҖ’еҪ’ең°жүҫеҮәе…ҘеҸЈж–Үд»¶зҡ„жүҖжңүдҫқиө–пјҢе°ҶвјҠеҸЈе’Ңе…¶жүҖжңүзҡ„дҫқиө–пјҢжү“еҢ…еҲ°дёҖдёӘеҚ•зӢ¬зҡ„ж–Үд»¶дёӯпјҢжҳҜе·ҘзЁӢеҢ–гҖҒиҮӘеҠЁеҢ–жҖқжғіеңЁеүҚз«ҜејҖеҸ‘дёӯзҡ„дҪ“зҺ°гҖӮ

дёҖеҲҮд»Ҙе®ҳвҪ…зҪ‘з«ҷдёәеҮҶпјҡhttps://webpack.js.org/

2.е®үиЈ…webpack¶

зҺҜеўғеҮҶеӨҮпјҡйңҖиҰҒе®үиЈ… nodeJs https://nodejs.org/en/пјҢзүҲжң¬еҸӮиҖғе®ҳвҪ№еҸ‘еёғзҡ„жңҖж–°зүҲжң¬пјҢж–°зүҲжң¬еҸҜд»ҘжҸҗеҚҮwebpackзҡ„жү“еҢ…йҖҹеәҰ

е…ЁеұҖе®үиЈ… пҘ§жҺЁиҚҗ 

# е®үиЈ…webpack V4+зүҲжң¬ж—¶пјҢйңҖиҰҒйўқеӨ–е®үиЈ…webpack-cli
npm install webpack webpack-cli -g
webpack -v
npm uninstall webpack webpack-cli -g

е…ЁеұҖе®үиЈ…webpackпјҢиҝҷдјҡе°ҶдҪ йЎ№вҪ¬дёӯзҡ„webpackй”Ғе®ҡеҲ°жҢҮе®ҡзүҲжң¬пјҢйҖ жҲҗдёҚеҗҢзҡ„йЎ№вҪ¬дёӯеӣ дёәwebpackдҫқиө–дёҚеҗҢзүҲжң¬вҪҪиҖҢеҜјиҮҙеҶІзӘҒпјҢжһ„е»әеӨұиҙҘгҖӮ

йЎ№вҪ¬е®үиЈ… жҺЁиҚҗ

npm install webpack webpack -D

# е®үиЈ…жңҖж–°зҡ„зЁіе®ҡзүҲжң¬
npm i -D webpack

# е®үиЈ…жҢҮе®ҡзүҲжң¬
npm i -D webpack@<version>

# е®үиЈ…жңҖж–°зҡ„дҪ“йӘҢзүҲжң¬ еҸҜиғҪеҢ…еҗ«bug,дёҚиҰҒз”ЁдәҺвҪЈз”ҹдә§зҺҜеўғ
npm i -D webpack@beta

# е®үиЈ…webpack V4+зүҲжң¬ж—¶пјҢйңҖиҰҒйўқеӨ–е®үиЈ…webpack-cli
npm i -D webpack-cli

жЈҖжҹҘе®үиЈ…

webpack -v 
# еҸҜиғҪеҮәзҺ°жҠҘй”ҷпјҡcommand not found й»ҳи®ӨеңЁе…ЁеұҖзҺҜеўғдёӯжҹҘжүҫпјҢе…ЁеұҖзҺҜеўғдёӢйқўжІЎжңүе®үиЈ… webpack

3.еҗҜеҠЁwebpackжү§вҫҸжһ„е»ә¶

1Chunk = 1bundle

chunk жҳҜиҷҡжӢҹзҡ„д»Јз Ғеқ—

bundle жҳҜз”ҹжҲҗзҡ„иө„жәҗвҪӮд»¶

1дёӘchunkеҸҜд»ҘжҳҜеӨҡдёӘжЁЎеқ—з»„жҲҗзҡ„

жЁЎеқ—пјҡnodeJSвҫҘ дёҮзү©зҡҶжЁЎеқ—

entry зҡ„ж•°жҚ®зұ»еһӢпјҡеӯ—з¬ҰдёІгҖҒж•°з»„гҖҒеҜ№иұЎгҖӮ

еӯ—з¬ҰдёІе’Ңж•°з»„зҡ„ж•Ҳжһңеҹәжң¬дёҖж ·пјҢжңҖеҗҺйғҪжҳҜеҚ•дёҖе…ҘеҸЈгҖӮй»ҳи®ӨжҳҜеӯ—з¬ҰдёІгҖӮ

еҰӮжһң entry жҳҜж•°з»„: webpackдјҡвҫғеҠЁвҪЈз”ҹжҲҗеҸҰеӨ–вјҖдёӘе…ҘвјқжЁЎеқ—пјҢе№¶е°Ҷж•°з»„дёӯзҡ„жҜҸдёӘжҢҮе®ҡзҡ„жЁЎеқ—еҠ иҪҪиҝӣжқҘпјҢе№¶е°ҶжңҖеҗҺвјҖдёӘжЁЎеқ—зҡ„ module.exports дҪңдёәе…ҘвјқжЁЎеқ— зҡ„module.exportsеҜјеҮәгҖӮ

еҰӮжһң entry жҳҜеҜ№иұЎеӨҡй”®еҖјпјҢеҜ№еә”еӨҡдёӘ chunkпјҢжүҚжҳҜеӨҡдёӘжү“еҢ…пјҢеӨҡйЎөйқўеә”з”ЁгҖӮ

3.1- webpackй»ҳи®Өй…ҚзҪ®¶

  • webpackй»ҳи®ӨвҪҖжҢҒJSжЁЎеқ—е’ҢJSONжЁЎеқ—
  • ж”ҜжҢҒCommonJS Es moudule AMD CMDзӯүжЁЎеқ—зұ»еһӢ
  • webpack4вҪҖжҢҒйӣ¶й…ҚзҪ®дҪҝвҪӨ,дҪҶжҳҜеҫҲејұпјҢзЁҚеҫ®еӨҚжқӮдәӣзҡ„еңәжҷҜйғҪйңҖиҰҒ йўқеӨ–жү©еұ•

3.2- еҮҶеӨҮжү§вҫҸжһ„е»ә¶

ж–°е»әsrcвҪӮд»¶еӨ№пјҢж–°е»әsrc/index.jsгҖҒsrc/index.jsonгҖҒsrc/other.js

// index.js
const json = require("./index.json"); //commonJS
import { add } from "./other.js"; //es module
console.log(json, add(2, 3));

index.json

{
  "name": "JOSN"
}

// other.js
export function add(n1, n2) {
  return n1 + n2;
}

npx webpack

"scripts": {

дҝ®ж”№package.jsonвҪӮд»¶:

еҺҹзҗҶе°ұжҳҜйҖҡиҝҮshellи„ҡжң¬еңЁ node_modules/.binвҪ¬еҪ•дёӢеҲӣе»әвјҖдёӘиҪҜй“ҫжҺҘгҖӮ

3.4-жһ„е»әжҲҗеҠҹ¶

жҲ‘д»¬дјҡеҸ‘зҺ°вҪ¬еҪ•дёӢеӨҡеҮәвјҖдёӘ dist вҪ¬еҪ•пјҢвҫҘвҫҜжңүдёӘ main.js пјҢиҝҷдёӘвҪӮд»¶жҳҜвјҖдёӘеҸҜжү§вҫҸзҡ„JavaScriptвҪӮд»¶пјҢвҫҘвҫҜеҢ…еҗ« webpackBootstrap еҗҜеҠЁеҮҪж•°гҖӮ

3.5-й»ҳи®Өй…ҚзҪ®¶

const path = require("path");
module.exports = {
  // еҝ…еЎ« webpackжү§вҫҸжһ„е»әе…Ҙвјқ 
  entry: "./src/index.js", 
  output: {
    // е°ҶжүҖжңүдҫқиө–зҡ„жЁЎеқ—еҗҲе№¶иҫ“еҮәеҲ°main.js 
    filename: "[name].[hash6].js",
    // иҫ“еҮәвҪӮд»¶зҡ„еӯҳж”ҫи·ҜпӨ·еҫ„пјҢеҝ…йЎ»жҳҜз»қеҜ№и·ҜпӨ·еҫ„
        path: path.resolve(__dirname, "./dist")
    }
};

4.webpackй…ҚзҪ®ж ёвјјжҰӮеҝө¶

йӣ¶й…ҚзҪ®жҳҜеҫҲејұзҡ„пјҢзү№е®ҡзҡ„йңҖжұӮпјҢжҖ»жҳҜйңҖиҰҒвҫғвј°е·ұиҝӣвҫҸй…ҚзҪ®

webpackжңүй»ҳи®Өзҡ„й…ҚзҪ®вҪӮд»¶пјҢеҸ« webpack.config.js пјҢжҲ‘д»¬еҸҜд»ҘеҜ№иҝҷдёӘвҪӮд»¶иҝӣвҫҸдҝ®ж”№пјҢиҝӣвҫҸдёӘжҖ§еҢ–й…ҚзҪ®

дҪҝвҪӨй»ҳи®Өзҡ„й…ҚзҪ®вҪӮд»¶:webpack.config.js

дёҚдҪҝвҪӨвҫғе®ҡд№үй…ҚзҪ®вҪӮд»¶: вҪҗеҰӮwebpackconfig.jsпјҢеҸҜд»ҘйҖҡиҝҮ-- config webpackconfig.jsжқҘжҢҮе®ҡwebpackдҪҝвҪӨе“ӘдёӘй…ҚзҪ®вҪӮд»¶жқҘ жү§вҫҸжһ„е»ә

webpack.config.jsй…ҚзҪ®еҹәзЎҖз»“жһ„

module.exports = {
  entry: "./src/index.js", //жү“еҢ…е…ҘвјқвҪӮд»¶
  output: "./dist", //иҫ“еҮәз»“жһ„
  mode: "production", //жү“еҢ…зҺҜеўғ
  module: {
    rules: [ //loaderжЁЎеқ—еӨ„зҗҶ
      {
        test: /\.css$/,
        use: "style-loader"
      }
    ]
  },
  //жҸ’д»¶й…ҚзҪ® 
  plugins: [
    new HtmlWebpackPlugin()
  ]
};

4.1-entry:¶

жҢҮе®ҡwebpackжү“еҢ…е…ҘвјқвҪӮд»¶:Webpack жү§вҫҸжһ„е»әзҡ„з¬¬вјҖжӯҘе°Ҷд»Һ Entry ејҖе§ӢпјҢеҸҜжҠҪиұЎжҲҗиҫ“е…Ҙ

//еҚ•е…Ҙвјқ SPAпјҢжң¬иҙЁжҳҜдёӘеӯ—з¬ҰдёІ 
entry:{
  main: './src/index.js'
}
// ==зӣёеҪ“дәҺз®ҖеҶҷ=== entry:"./src/index.js"

//еӨҡе…Ҙвјқ entryжҳҜдёӘеҜ№иұЎ
entry:{
  index:"./src/index.js",
  login:"./src/login.js"

        use: ["style-loader", "css-loader"]
      },
    ],
  },
};
