    LDAPе…Ҙй—Ё¶

    йҰ–е…ҲLDAPжҳҜдёҖз§ҚйҖҡи®ҜеҚҸи®®пјҢLDAPж”ҜжҢҒTCP/IPгҖӮеҚҸи®®е°ұжҳҜж ҮеҮҶпјҢе№¶дё”жҳҜжҠҪиұЎзҡ„гҖӮеңЁиҝҷеҘ—ж ҮеҮҶдёӢпјҢADпјҲActive DirectoryпјүжҳҜеҫ®иҪҜеҮәзҡ„дёҖеҘ—е®һзҺ°гҖӮ

    йӮЈADжҳҜд»Җд№Ҳе‘ўпјҹжҡӮдё”жҠҠе®ғзҗҶи§ЈжҲҗжҳҜдёӘж•°жҚ®еә“гҖӮд№ҹжңүеҫҲеӨҡдәәзӣҙжҺҘжҠҠ LDAP иҜҙжҲҗж•°жҚ®еә“(еҸҜд»ҘжҠҠLDAPзҗҶи§ЈжҲҗеӯҳеӮЁж•°жҚ®зҡ„ж•°жҚ®еә“)гҖӮеғҸжҳҜе…¶д»–ж•°жҚ®еә“дёҖж ·пјҢLDAPд№ҹжҳҜжңүclientз«Ҝе’Ңserverз«ҜгҖӮserverз«ҜжҳҜз”ЁжқҘеӯҳж”ҫиө„жәҗпјҢclientз«Ҝз”ЁжқҘж“ҚдҪңеўһеҲ ж”№жҹҘзӯүж“ҚдҪңгҖӮ

    иҖҢжҲ‘д»¬йҖҡеёёиҜҙзҡ„LDAPжҳҜжҢҮиҝҗиЎҢиҝҷдёӘж•°жҚ®еә“зҡ„жңҚеҠЎеҷЁгҖӮеҸҜд»Ҙз®ҖеҚ•зҗҶи§Ј AD =LDAPжңҚеҠЎеҷЁпјӢLDAPеә”з”ЁгҖӮ

    йӮЈLDAPиҝҷз§Қж•°жҚ®еә“жңүд»Җд№Ҳзү№ж®Ҡзҡ„е‘ўпјҹжҲ‘д»¬зҹҘйҒ“пјҢеғҸMySQLж•°жҚ®еә“пјҢж•°жҚ®йғҪжҳҜжҢүи®°еҪ•дёҖжқЎжқЎи®°еҪ•еӯҳеңЁиЎЁдёӯгҖӮиҖҢ LDAP ж•°жҚ®еә“пјҢжҳҜж ‘з»“жһ„зҡ„пјҢж•°жҚ®еӯҳеӮЁеңЁеҸ¶еӯҗиҠӮзӮ№дёҠгҖӮзңӢзңӢдёӢйқўзҡ„жҜ”е–»пјҡ

    еҒҮи®ҫдҪ иҰҒж ‘дёҠзҡ„дёҖдёӘиӢ№жһңпјҲдёҖжқЎи®°еҪ•пјүпјҢдҪ жҖҺд№Ҳе‘ҠиҜүе®ғзҡ„дҪҚзҪ®е‘ўпјҹеҪ“з„¶йҰ–е…ҲиҰҒиҜҙжҳҺжҳҜе“ӘдёҖжЈөж ‘пјҲdcпјҢзӣёеҪ“дәҺMYSQLзҡ„DBпјүпјҢз„¶еҗҺжҳҜд»Һж ‘ж №еҲ°йӮЈдёӘиӢ№жһңжүҖз»ҸиҝҮзҡ„жүҖжңүвҖңеҲҶеҸүвҖқпјҲouпјүпјҢжңҖеҗҺе°ұжҳҜиҝҷдёӘиӢ№жһңзҡ„еҗҚеӯ—пјҲuidпјҢзӣёеҪ“дәҺMySQLиЎЁдё»й”®idпјүгҖӮеҘҪдәҶпјҒиҝҷж—¶жҲ‘д»¬еҸҜд»Ҙжё…жҷ°зҡ„жҢҮжҳҺиҝҷдёӘиӢ№жһңзҡ„дҪҚзҪ®дәҶгҖӮе°ұиҝҷж ·е°ұеҸҜд»ҘжҸҸиҝ°жё…жҘҡвҖңж ‘з»“жһ„вҖқдёҠзҡ„дёҖжқЎи®°еҪ•дәҶгҖӮ

    иҜҙдёҖдёӢLDAPйҮҢеҰӮдҪ•е®ҡд№үдёҖдёӘи®°еҪ•зҡ„дҪҚзҪ®еҗ§гҖӮ

    ж ‘пјҲdc=ljheee)
еҲҶеҸүпјҲou=bei,ou=xi,ou= dongпјү
иӢ№жһңпјҲcn=redAppleпјү

    еҘҪдәҶпјҢredAppleзҡ„дҪҚзҪ®еҮәжқҘдәҶпјҡ

     dn:cn=honglv,ou=bei,ou=xi,ou=dong,dc=ljheee

    е…¶дёӯdnж ҮиҜҶдёҖжқЎи®°еҪ•пјҢжҸҸиҝ°дәҶдёҖжқЎж•°жҚ®зҡ„иҜҰз»Ҷи·Ҝеҫ„гҖӮ

    е’Ұ!жңүдәәз–‘й—®пјҢдёәд»Җд№ҲouдјҡжңүеӨҡдёӘеҖјпјҹдҪ жғіжғіпјҢд»Һж ‘ж №еҲ°иҫҫиӢ№жһңзҡ„дҪҚзҪ®пјҢеҸҜиғҪиҰҒз»ҸиҝҮеҘҪеҮ дёӘж ‘жқҲпјҢжүҖжңүouеҸҜиғҪжңүеӨҡдёӘеҖјгҖӮе…ідәҺdnеҗҺйқўдёҖй•ҝдёІпјҢеҲҶеҲ«жҳҜcnпјҢou,dcпјӣдёӯй—ҙз”ЁйҖ—еҸ·йҡ”ејҖгҖӮ

    жҖ»з»“дёҖдёӢLDAPж ‘еҪўж•°жҚ®еә“еҰӮдёӢпјҡ

    dn пјҡдёҖжқЎи®°еҪ•зҡ„иҜҰз»ҶдҪҚзҪ®

    dc пјҡдёҖжқЎи®°еҪ•жүҖеұһеҢәеҹҹ (е“ӘдёҖйў—ж ‘)

    ou пјҡдёҖжқЎи®°еҪ•жүҖеұһз»„з»Ү пјҲе“ӘдёҖдёӘеҲҶж”Ҝпјү

    cn/uidпјҡдёҖжқЎи®°еҪ•зҡ„еҗҚеӯ—/ID (е“ӘдёҖдёӘиӢ№жһңеҗҚеӯ—)

    LDAPзӣ®еҪ•ж ‘зҡ„жңҖйЎ¶йғЁе°ұжҳҜж №пјҢд№ҹе°ұжҳҜжүҖи°“зҡ„вҖңеҹәеҮҶDN"гҖӮ

    дёәд»Җд№ҲиҰҒз”ЁLDAPзӣ®еҪ•ж ‘жқҘеӯҳеӮЁж•°жҚ®пјҢз”ЁMySQLдёҚиЎҢеҗ—пјҢдёәд»Җд№ҲйқһиҰҒжҗһеҮәдёҖдёӘж ‘еҪўзҡ„ж•°жҚ®еә“е‘ўпјҹ

    иҝҷжҳҜеӣ дёәз”Ёж ‘еҪўз»“жһ„еӯҳеӮЁж•°жҚ®пјҢжҹҘиҜўж•ҲзҺҮжӣҙй«ҳпјҲе…·дҪ“дёәд»Җд№ҲпјҢеҸҜд»ҘзңӢдёҖдёӢе…ізі»еһӢж•°жҚ®еә“зҙўеј•зҡ„е®һзҺ°еҺҹзҗҶвҖ”вҖ”Bж ‘/B+ж ‘пјүгҖӮеңЁжҹҗдәӣзү№е®ҡзҡ„еңәжҷҜдёӢпјҢдҪҝз”Ёж ‘еҪўж•°жҚ®еә“жӣҙзҗҶжғігҖӮжҜ”еҰӮпјҡйңҖиҰҒеӮЁеӯҳеӨ§йҮҸзҡ„ж•°жҚ®пјҢиҖҢдё”ж•°жҚ®дёҚжҳҜз»Ҹеёёжӣҙж”№пјҢйңҖиҰҒеҫҲеҝ«йҖҹзҡ„жҹҘжүҫгҖӮ

    жҠҠе®ғдёҺдј з»ҹзҡ„е…ізі»еһӢж•°жҚ®еә“зӣёжҜ”пјҢLDAPйҷӨдәҶеҝ«йҖҹжҹҘжүҫзҡ„зү№зӮ№пјҢе®ғиҝҳжңүеҫҲеӨҡзҡ„иҝҗз”ЁеңәжҷҜпјҢжҜ”еҰӮеҹҹйӘҢиҜҒзӯүгҖӮ

    LDAP javaзј–зЁӢж“ҚдҪң¶

    жҲ‘д»¬еҸҜд»Ҙз”Ё JDBC пјҲJava ж•°жҚ®еә“иҝһжҺҘпјҢJava Database ConnectivityпјүпјҢж“ҚдҪңMySQLж•°жҚ®еә“пјҢиҝӣиЎҢеҜ№ж•°жҚ®зҡ„еўһеҲ ж”№жҹҘгҖӮеҗҢж ·пјҢLDAPж ‘еҪўж•°жҚ®еә“пјҢд№ҹеҸҜд»ҘйҖҡиҝҮ JDBC ж–№ејҸпјӣйҷӨжӯӨд№ӢеӨ–пјҢиҝҳеҸҜд»Ҙз”Ё JNDI зҡ„ж–№ејҸпјҲJava Naming and Directory InterfaceпјҢJavaе‘ҪеҗҚе’Ңзӣ®еҪ•жҺҘеҸЈпјҢжӣҙжҺЁиҚҗпјүпјҢеӣ дёәж ‘еҪўеҸҜд»ҘзңӢеҒҡжҳҜзӣ®еҪ•пјҢж ‘з»“жһ„зҡ„жһқжқҲзӣёеҪ“дәҺзӣ®еҪ•зҡ„еұӮзә§гҖӮ

    иҝҳжңүLDAPж•°жҚ®еә“еұ•зӨәж•°жҚ®д№ҹжҳҜж ‘еҪўзҡ„пјҢApacheDirectoryStudioдёӢиҪҪең°еқҖ

    http://download.csdn.net/download/ljheee/10145654

    Javaе‘ҪеҗҚе’Ңзӣ®еҪ•жҺҘеҸЈпјҲдәҶи§Јпјү

    import com.xxx.csb.ldap.config.LdapConfiguration;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.Hashtable;
import javax.naming.*;
import javax.naming.directory.Attribute;
import javax.naming.directory.Attributes;
import javax.naming.directory.DirContext;
import javax.naming.directory.InitialDirContext;

public class LdapJNDI {

  @Autowired
  LdapConfiguration ldapConfiguration;

  public void JNDILookup() {
    String rootFilter = "o=xxx.com,o=isp";
    String filter = "(&(smart-type=E1)(smart-status=1))";
    String filter = "(&(smart-type=E1)(uid=00012047))";
    String username = "uid=USER_NAME,ou=Authorization,ou=People,o=cc.com,o=isp";//xxxдёәз”іиҜ·зҡ„еҜ№жҺҘиҙҰжҲ·
    String password = "PASSW";

    Hashtable env = new Hashtable();
    env.put(Context.INITIAL_CONTEXT_FACTORY, "com.sun.jndi.ldap.LdapCtxFactory");//и®ҫзҪ®иҝһжҺҘLDAPзҡ„е®һзҺ°е·ҘеҺӮ
    env.put(Context.PROVIDER_URL, "ldap://172.16.32.19:389/" + rootFilter);// жҢҮе®ҡLDAPжңҚеҠЎеҷЁзҡ„дё»жңәеҗҚе’Ңз«ҜеҸЈеҸ·
    env.put(Context.SECURITY_AUTHENTICATION, "simple");//з»ҷзҺҜеўғжҸҗдҫӣи®ӨиҜҒж–№жі•,жңүSIMPLEгҖҒSSL/TLSе’ҢSASL
    env.put(Context.SECURITY_PRINCIPAL, username);//жҢҮе®ҡиҝӣе…Ҙзҡ„зӣ®еҪ•иҜҶеҲ«еҗҚDN
    env.put(Context.SECURITY_CREDENTIALS, password); //иҝӣе…Ҙзҡ„зӣ®еҪ•еҜҶз Ғ
    env.put("filter",filter);
    DirContext ctx = null;

    try {
      // еҫ—еҲ°еҲқе§Ӣзӣ®еҪ•зҺҜеўғзҡ„дёҖдёӘеј•з”Ё
      ctx = new InitialDirContext(env);
      //The search base entry 'uid=00012047,ou=Internal,ou=People,o=xxx.com,o=isp' does not exist]; remaining name 'uid=00012047,ou=Internal'
      Attributes attrs = ctx.getAttributes("uid=00012047,ou=Internal,ou=People");//иҺ·еҸ–еҲ°дёҖдёӘдәәе‘ҳпјҢ
      NamingEnumeration bindings = ctx.listBindings("ou=Internal,ou=People");//еҲ—дёҫ еҶ…йғЁдәәе‘ҳ
      while (bindings.hasMore()) {
        Binding bd = (Binding)bindings.next();
        System.out.println(bd.getName() + ": " + bd.getObject());
      }
    } catch (javax.naming.AuthenticationException e) {
      System.out.println("и®ӨиҜҒеӨұиҙҘ");
      e.printStackTrace();
    } catch (Exception e) {
      System.out.println("и®ӨиҜҒеҮәй”ҷпјҡ");
      e.printStackTrace();
    }finally {
      if (ctx != null) {
        try {
          ctx.close();
        } catch (NamingException e) {
          e.printStackTrace();
        }
      }
    }
  }

  public static void main(String[] args) {
    LdapJNDI ldapJNDI = new LdapJNDI();
    ldapJNDI.JNDILookup();
  }
}

//еҲӣе»әLDAP clientпјҢжҠҠжңҚеҠЎеҷЁurlдј е…Ҙ
var client = ldap.createClient({
  url: 'ldap://10.203.24.216:389'
});

//еҲӣе»әLDAPжҹҘиҜўйҖүйЎ№
//filterзҡ„дҪңз”Ёе°ұжҳҜзӣёеҪ“дәҺSQLзҡ„жқЎд»¶
var opts = {
  filter: '(uid=kxh)', //жҹҘиҜўжқЎд»¶иҝҮж»ӨеҷЁпјҢжҹҘжүҫuid=kxhзҡ„з”ЁжҲ·иҠӮзӮ№
  scope: 'sub',        //жҹҘиҜўиҢғеӣҙ
  timeLimit: 500       //жҹҘиҜўи¶…ж—¶
};

//е°Ҷclientз»‘е®ҡLDAP Server
//з¬¬дёҖдёӘеҸӮж•°пјҡжҳҜз”ЁжҲ·пјҢеҝ…йЎ»жҳҜд»Һж №иҠӮзӮ№еҲ°з”ЁжҲ·иҠӮзӮ№зҡ„е…Ёи·Ҝеҫ„
//з¬¬дәҢдёӘеҸӮж•°пјҡз”ЁжҲ·еҜҶз Ғ
client.bind('uid=supbind,cn=users,dc=tiger,dc=com', '123456', function (err, res1) {

  //ејҖе§ӢжҹҘиҜў
  //з¬¬дёҖдёӘеҸӮж•°пјҡжҹҘиҜўеҹәзЎҖи·Ҝеҫ„пјҢд»ЈиЎЁеңЁжҹҘиҜўз”ЁжҲ·дҝЎеҝғе°ҶеңЁиҝҷдёӘи·Ҝеҫ„дёӢиҝӣиЎҢпјҢиҝҷдёӘи·Ҝеҫ„жҳҜз”ұж №иҠӮејҖе§Ӣ
  //з¬¬дәҢдёӘеҸӮж•°пјҡжҹҘиҜўйҖүйЎ№
  client.search('DC=tiger,DC=com', opts, function (err, res2) {

    //жҹҘиҜўз»“жһңдәӢд»¶е“Қеә”
    res2.on('searchEntry', function (entry) {

      //иҺ·еҸ–жҹҘиҜўзҡ„еҜ№иұЎ
      var user = entry.object;
      var userText = JSON.stringify(user,null,2);
      console.log(userText);

    });

    res2.on('searchReference', function(referral) {
      console.log('referral: ' + referral.uris.join());
    });    

    //жҹҘиҜўй”ҷиҜҜдәӢд»¶
    res2.on('error', function(err) {
      console.error('error: ' + err.message);
      //unbindж“ҚдҪңпјҢеҝ…йЎ»иҰҒеҒҡ
      client.unbind();
    });

    //жҹҘиҜўз»“жқҹ
    res2.on('end', function(result) {
      console.log('search status: ' + result.status);
      //unbindж“ҚдҪңпјҢеҝ…йЎ»иҰҒеҒҡ
      client.unbind();
    });
  });
});
